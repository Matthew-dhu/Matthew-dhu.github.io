外部排序-多路归并

本文来自我们研究生课程的一个大作业。网上有一些c实现的版本，本文使用Java完成，中途遇到了一些问题，给需要的同学们一些参考。

需求如下

实验目的：通过merge-sort算法的实现，掌握外存算法所基于的I/O模型与内存算法基于的RAM模型的区别；理解不同的磁盘访问优化方法是如何提高数据访问性能的。

实验内容：生成一个具有10,000,000个记录的文本文件，其中每个记录由100个字节组成。实验只考虑记录的一个属性A，假定A为整数类型。记录在block上封装时，采用non-spanned方式，即块上小于一个记录的空间不使用。Block的大小可在自己的操作系统上查看，xp一般为4096 bytes。在内存分配50M字节的空间用于外部merge-sort。要求设计和实现程序完成下列功能：

主要困难在与Java对io的操作。由于一般直接存取int类型需要用InputStream和OutputStream的readInt和writeInt方法，但是这两个流没有设置缓冲区，大量数据存取时速度非常慢，而如果使用缓冲流，每次只能存取一个字节，在存取int时会自动将低位截掉。我的解决方法为：将int拆分成一个大小为4的char数组，然后进行存取。

需要注意的是，实验结果使用4k的buffer所用的时间比使用1M的buffer短，与预期的不符，经过查阅资料以及老师的提醒，可能是由于Java对文件的操作进行了封装，底层实现对我们不可见。

本实验的代码如下。

1.转换的工具类

public class IntByteArrayTransfer {
    public static byte[] intToByteArray(int i) {
        byte[] result = new byte[4];
        // 由高位到低位
        result[0] = (byte) ((i >> 24) & 0xFF);
        result[1] = (byte) ((i >> 16) & 0xFF);
        result[2] = (byte) ((i >> 8) & 0xFF);
        result[3] = (byte) (i & 0xFF);
        return result;
    }

    public static int byteArrayToInt(byte[] bytes) {
        int value = 0;
        // 由高位到低位
        for (int i = 0; i < 4; i++) {
        int shift = (4 - 1 - i) * 8;
        value += (bytes[i] & 0x000000FF) << shift;// 往高位游
        }
        return value;
    }

}

2.用到的常数

public class Constant {
    public static int NUMBER_OF_RECORDS=10000000;
    public static String FILE_NAME="D:\\实验数据\\data.txt";
    public static String FILE_DIR="D:\\实验数据\\";
    public static int RECORDS_PER_SUB_FILE=500000;
    public static int NUMBER_OF_SUB_FILE=20;
    public static int INPUT_BUFFER_SIZE=10000;
    public static int OUTPUT_BUFFER_SIZE=200000;
}

3.数据项

public class Record implements Comparable<Record>{
    //实现Comparable接口，便于后面使用Arrays.sort内部排序方法
    private int key;
    private String value;

    public int getKey() {
        return key;
    }

    public void setKey(int key) {
        this.key = key;
    }

    public String getValue() {
        return value;
    }

    public void setValue(String value) {
        this.value = value;
    }

    @Override
    public String toString() {
        return "Record{" +
                "key=" + key +
                '}';
    }

    @Override
    public int compareTo(Record record) {
        if(record.key==this.key)
            return 0;
        if(record.key>this.key){
            return -1;
        }
        return 1;
    }
}

4.生成无序文件及读取Record的类

public class Generate {
    //读一条数据
    public static Record readRecord(BufferedInputStream bis) throws IOException {
        Record record=new Record();
        byte[] buffer1=new byte[4];
        if(bis.read(buffer1,0,1)==-1){//判断是否读完
            return null;
        }
        bis.read(buffer1, 1, 3);
        record.setKey(IntByteArrayTransfer.byteArrayToInt(buffer1));
        byte[] buffer2=new byte[96];
        bis.read(buffer2, 0, 96);
        record.setValue(new String(buffer2,0,96));
        return record;
    }

    //写一条数据
    public static void writeRecord(BufferedOutputStream bos,Record record) throws IOException {
        bos.write(IntByteArrayTransfer.intToByteArray(record.getKey()));
        bos.write(record.getValue().getBytes());
    }

    //第一阶段输出1G随机数据的方法
    public static void writeFile(String fileName) {
        BufferedOutputStream bos=null;
        File file=new File(fileName);
        Random random=new Random();
        try {
            bos=new BufferedOutputStream(new FileOutputStream(file,true));
            for(int i=0;i<Constant.NUMBER_OF_RECORDS;i++){
                bos.write(IntByteArrayTransfer.intToByteArray(random.nextInt()));
                //因为int四个字节，而BufferedOutputStream每次读取一个字节，写int时会截掉低位，
                //而如果使用DataOutputStream因为每次都要访问io，速度极慢，所以将int拆分为一个四个元素的byte数组
                for(int j=0;j<96;j++){
                    bos.write('*');//*占一个字节
                }
            }
        } catch (IOException e) {
            e.printStackTrace();
        } finally {
            try {
                if(bos!=null){
                    bos.close();
                }
            }catch (Exception  e){
                e.printStackTrace();
            }
        }
    }
}

5.两个阶段

public class Phases {
    public static void phase1(){
        BufferedInputStream bis=null;
        File file2=new File(Constant.FILE_NAME);
        BufferedOutputStream bos=null;
        try {
            bis=new BufferedInputStream(new FileInputStream(file2));
            for(int i=0;i<Constant.NUMBER_OF_SUB_FILE;i++){
                Record[] records=new Record[Constant.RECORDS_PER_SUB_FILE];
                File file=new File(Constant.FILE_DIR,"subData_"+i+".txt");
                if(!file.exists()){
                        file.createNewFile();
                }
                for(int j=0;j<Constant.RECORDS_PER_SUB_FILE;j++){
                    records[j]=Generate.readRecord(bis);
                }
                Arrays.sort(records);

                //写到文件中
                bos=new BufferedOutputStream(new FileOutputStream(file,true));
                for(int k=0;k<records.length;k++){
                    bos.write(IntByteArrayTransfer.intToByteArray(records[k].getKey()));
                    bos.write(records[k].getValue().getBytes());
                }
                bos.flush();
            }
        } catch (IOException e) {
            e.printStackTrace();
        }finally {
            try {
                if(bos!=null){
                    bos.close();
                }
                if(bis!=null){
                    bis.close();
                }
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }

    public static void phase2(){
        BufferedInputStream[] biss=new BufferedInputStream[Constant.NUMBER_OF_SUB_FILE];
        Record[][] inputBuffer=new Record[Constant.NUMBER_OF_SUB_FILE][Constant.INPUT_BUFFER_SIZE];
        Record[] outputBuffer=new Record[Constant.OUTPUT_BUFFER_SIZE];
        BufferedOutputStream bos=null;
        File file=new File(Constant.FILE_DIR+"sortedData.txt");
        int outNum=0;//记录输出buffer写到的位置
        List<Integer> list=new ArrayList<>();//记录可读的输入buffer
        for(int i=0;i<Constant.NUMBER_OF_SUB_FILE;i++){
            list.add(i);
        }
        int[] count=new int[Constant.NUMBER_OF_SUB_FILE];//记录每个输入buffer读到的位置
        for(int i=0;i<Constant.NUMBER_OF_SUB_FILE;i++){
            count[i]=0;
        }
        try {
            if(!file.exists()){
                file.createNewFile();
            }
            bos=new BufferedOutputStream(new FileOutputStream(file,true));
            for(int i=0;i<Constant.NUMBER_OF_SUB_FILE;i++){
                biss[i]=new BufferedInputStream(new FileInputStream(Constant.FILE_DIR+"subData_"+i+".txt"));
                for(int j=0;j<Constant.INPUT_BUFFER_SIZE;j++){
                    inputBuffer[i][j]= Generate.readRecord(biss[i]);
                }
            }
            while(!list.isEmpty()){
                Integer min=null;
                Integer minIndex=null;
                for(int i:list){//找到最小值
                    if(min==null||min>inputBuffer[i][count[i]].getKey()){
                        min=inputBuffer[i][count[i]].getKey();
                        minIndex=i;
                    }
                }
                if(outNum==Constant.OUTPUT_BUFFER_SIZE){//如果输出buffer满全部写入文件中
                    for(int i=0;i<Constant.OUTPUT_BUFFER_SIZE;i++){
                        Generate.writeRecord(bos,outputBuffer[i]);
                    }
                    bos.flush();
                    outNum=0;
                }
                outputBuffer[outNum]=inputBuffer[minIndex][count[minIndex]];
                outNum++;
                count[minIndex]++;
                if(count[minIndex]==Constant.INPUT_BUFFER_SIZE){//如果输入buffer满
                    if((inputBuffer[minIndex][0]=Generate.readRecord(biss[minIndex]))==null){
                        list.remove(Integer.valueOf(minIndex));
                    }else {
                        for(int i=1;i<Constant.INPUT_BUFFER_SIZE;i++){
                            inputBuffer[minIndex][i]= Generate.readRecord(biss[minIndex]);
                        }
                        count[minIndex]=0;
                    }
                }
            }
            //特别注意！最后有剩余的数据没有写入文件
            for(int i=0;i<outNum;i++){
                Generate.writeRecord(bos,outputBuffer[i]);
            }
            bos.flush();
        }catch (IOException e){
            e.printStackTrace();
        }finally {
            try {
                for(int i=0;i<biss.length;i++){
                    if(biss[i]!=null){
                        biss.clone();
                    }
                }
                if(bos!=null){
                    bos.close();
                }
            }catch (IOException e) {
                e.printStackTrace();
            }
        }

    }
}
